# Inventory_Manager
Built with Django, a work in progress.

DEPENDENCIES:
crispy forms : pip install crispy_forms,
crispy bootstrap5 : pip install crispy_bootstrap5,
plotly : pip install plotly,
pandas : pip install django-pandas,
